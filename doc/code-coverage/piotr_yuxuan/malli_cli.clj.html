011  (defn children-successor
012    "Given a schema `[:enum :a :b :c :d]`, return a Clojure map (that is,
013    a queryable data structure) that returns the next item of
014    :a -> :b -> :c -> :d <-> :d. The last item is mapped onto itself. If
015    the schema has a default value as a property, like in
016    `[:enum {:default :b} :a :b :c :d]` an additional mapping will be
017    made nil -> :c.
018  
019    Primarily intended to be used on enum schema for non-idempotent
020    options (like :verbose), but code is generic so you might think of a
021    use case on another schema type."
022    [schema]
023    (let [properties (m/properties schema)
024          children (m/children schema)
025          last-child (last children)]
026      (cond-> {}
027        :item->successor (into (map vec (partition 2 1 children)))
028        ;; The transformer `mt/default-value-transformer` is only
029        ;; applied once the args vector is decoded into a map, so we
030        ;; have to handle it manually.
031        (contains? properties :default) (as-> $ (assoc $ nil (get $ (:default properties))))
032        :loop-on-last (assoc last-child last-child))))
033  
034  (def non-idempotent-option
035    (fn [options {:keys [in schema]} _cli-args]
036      (update-in options in (children-successor schema))))

076  (defn label+value-schema
077    "Return `MapEntry` items, when applicable one for short, and long
078    option names."
079    [{:keys [in schema] :as value-schema}]
080    (let [in' (remove #(and (keyword? %) (= (namespace %) "malli.core")) in)
081          default-label (->> in' (mapcat name-items) (str/join "-"))
082          value-schema' (-> value-schema
083                            (merge (m/type-properties schema) (m/properties schema))

085      [(long-option->value-schema default-label value-schema')
086       (short-option->value-schema default-label value-schema')]))

091  (defn ^ParsingResult -parse-option
092    "Take the current arglist head `arg`, the tail args-tail`. Depending
093    on the value schema consume some items from the tail and when
094    applicable pass them on to `update-fn`. This is actually the core of
095    the work that transforms a vector of string to a map of options."
096    [{:keys [in update-fn arg-number schema] :as value-schema} options arg argstail]
097    (cond (and update-fn (not arg-number)) (ParsingResult. (update options ::schema-errors conj {:message "update-fn needs arg-number", :arg arg, :schema schema})
098                                                           argstail)
099          (and update-fn arg-number) (ParsingResult. (update-fn options value-schema (take arg-number argstail))
100                                                     (drop arg-number argstail))
101          (zero? arg-number) (ParsingResult. (assoc-in options in true)
102                                           argstail)
103          (= 1 arg-number) (ParsingResult. (assoc-in options in (first argstail))
104                                           (rest argstail))
105          (number? arg-number) (ParsingResult. (assoc-in options in (vec (take arg-number argstail)))
106                                               (drop arg-number argstail))
107          :generic-error (ParsingResult. (update options ::schema-errors conj {:message "generic error", :arg arg, :schema schema})
108                                         argstail)))

110  (defn break-short-option-group
111    "Expand a group of short option labels into a several short labels and
112    interpolate them with the tail of the arglist args-tail` depending
113    on the number of arguments each option needs. "
114    [label+value-schemas arg argstail]
115    (loop [[{:keys [arg-number short-option] :as value-schema} & ss] (->> (rest arg)
116                                                                          (map #(str "-" %))
117                                                                          (map label+value-schemas))
118           interpolated-args ()
119           argstail argstail]
120      (if (nil? value-schema)
121        (into argstail interpolated-args)
122        (recur ss
123               (into (cons short-option interpolated-args) (take arg-number argstail))
124               (drop arg-number argstail)))))

132  (defn parse-args
133    "Entry point to the technical work of turning a sequence of arguments
134    `args` into a map that (maybe) conforms to a `schema`. It returns a
135    map of the options as parsed according to the schema, but with two
136    additional keys:
137  
138    - `::operands` is a vector of application arguments, that is to say
139      command-line arguments that do not represent an option value.
140  
141    - `::cli-args` is the raw, untouched vector of command-line
142      arguments received as input. Perhaps you need it for some
143      additional validation of positional logic."
144    [label+value-schemas args]
145    ;; TODO Validate assumption on schema.
146    (loop [options {}
147           operands []
148           [arg & argstail] args]
149      (cond
150        (nil? arg) ; Argument list to parse is exhausted
151        (assoc options
152          ::operands operands
153          ::cli-args args)
154  
155        (= posix/option-terminator arg)
156        (recur options (into operands argstail) [])
157  
158        (gnu/long-option-with-value? arg)
159        (recur options
160               operands
161               (break-long-option-and-value arg argstail))
162  
163        (and (get label+value-schemas arg)
164             (or (gnu/long-option-without-value? arg)
165                 (posix/single-option-without-value? arg)))
166        (let [parsing-result (-parse-option (get label+value-schemas arg) options arg argstail)]
167          (recur (.-options parsing-result)
168                 operands
169                 (.-argstail parsing-result)))
170  
171        (or (gnu/long-option-without-value? arg)
172            (posix/single-option-without-value? arg))
173        (recur (-> options
174                   (update ::unknown-option-errors conj {:arg arg})
175                   (assoc ::known-options (keys label+value-schemas)))
176               operands
177               argstail)
178  
179        (posix/grouped-options? arg)
180        (recur options
181               operands
182               (break-short-option-group label+value-schemas arg argstail))
183  
184        :operand
185        (recur options
186               (conj operands arg)
187               argstail))))

189  (def args-transformer
190    "The malli transformer wrapping `parse-args`. To be used it with
191    `m/decode`, wrapped by `mt/transformer`. Merely turn a sequence of
192    arguments `args` into a map that (maybe) conforms to a `schema`. You
193    can compose this transformer to further refine command-line argument
194    parsing. See `simple-cli-transformer` for an example."
195    {:name :args-transformer
196     :compile (fn [schema _]
197                (let [label+value-schemas (->> (m'/value-schemas schema)
198                                               (mapcat label+value-schema)
199                                               (into {}))]
200                  (fn [args]
201                    (parse-args label+value-schemas
202                                args))))})

204  (def cli-transformer
205    "Use it for dumb, do-what-I-mean cli args parsing. Simple transformer
206    for the most common use cases when you only want to get a (nested)
207    map of options out of command-line arguments:
208  
209    - `mt/strip-extra-keys-transformer` will remove any extraneous keys;
210  
211    - `m'/default-value-transformer` with `:env-var` injects environment
212      variables (read at decode time);
213  
214    - `(mt/default-value-transformer {:key :default})` fills the blank
215      with default values when applicable."
216    (mt/transformer
217      args-transformer
218      mt/strip-extra-keys-transformer ; Remove it for debug, or more advanced usage.
219      mt/string-transformer
220      (m'/default-value-transformer {:key :env-var
221                                     :default-fn #(System/getenv %)})
222      (mt/default-value-transformer {:key :default})))

230  (defn prefix-shadowing
231    [value-schemas]
232    (loop [[{:keys [schema] :as head} & tail] value-schemas
233           known-prefix? #{}
234           remainder value-schemas]
235      (let [prefix (some-> schema m/properties :summary-path)]
236        (cond (nil? head) remainder
237              (and prefix (known-prefix? prefix)) (recur tail known-prefix? remainder)
238              prefix (recur tail
239                            (conj known-prefix? prefix)
240                            (remove (comp (partial start-with? prefix) :path) remainder))
241              :else (recur tail known-prefix? remainder)))))

253  (defn summary
254    [schema]
255    (let [short-option-name (comp first second)
256          long-option-name (comp first first)
257          description (comp #(->> % :description (:summary %)) second first)
258          summary-table (->> (m'/value-schemas schema)
259                             prefix-shadowing
260                             (map label+value-schema)
261                             ;; All through `str` so that nil is rendered as empty string.
262                             (map (juxt (comp str short-option-name)
263                                        (comp str long-option-name)
264                                        (comp str default-value)
265                                        (comp str description)))
266                             (cons summary-header))
267          max-column-widths (reduce (fn [acc row] (map (partial max) (map count row) acc))
268                                    (repeat 0)
269                                    summary-table)]
270      (str/join "\n" (map (fn [v]
271                            (let [fmt (-make-format max-column-widths)]
272                              (.stripTrailing ^String (apply format fmt v))))
273                          summary-table))))
