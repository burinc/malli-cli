009  (defn -make-format
010    ;; From clojure/tools.cli
011    "Given a sequence of column widths, return a string suitable for use in
012    format to print a sequences of strings in those columns."
013    [lens]
014    (str/join (map #(str "  %" (when-not (zero? %) (str "-" %)) "s") lens)))
015  
016  (defn deep-merge
017    "It merges maps recursively. It merges the maps from left
018    to right and the right-most value wins. It is useful to merge the
019    user defined configuration on top of the default configuration.
020    example:
021    ``` clojure
022    (deep-merge {:foo 1 :bar {:baz 2}}
023                {:foo 2 :bar {:baz 1 :qux 3}})
024    ;;=> {:foo 2, :bar {:baz 1, :qux 3}}
025    ```
026    From https://github.com/BrunoBonacci/1config"
027    [& maps]
028    (let [maps (filter (comp not nil?) maps)]
029      (if (every? map? maps)
030        (apply merge-with deep-merge maps)
031        (last maps))))
